CREATE TABLE categories(
    CategoryID INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Title varchar(30),
    Description varchar(30)
);
CREATE TABLE businesses(
    BusinessID INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Name varchar(30),
    Address varchar(30),
    City varchar(30),
    Telephone varchar(30),
    URL varchar(30)
);
CREATE TABLE biz_categories(
    BusinessID INTEGER REFERENCES businesses(BusinessID),  
    CategoryID INTEGER REFERENCES categories(CategoryID)
);
ALTER TABLE biz_categories MODIFY CategoryID varchar(10);
ALTER TABLE categories MODIFY CategoryID varchar(10);
